PROJECTILE "Meilin Knifes Red"
	set color "blue"
	randomColor
	set velocity 100
	inc velocityAngle 175
	set acceleration -genTime * 10 * random
	inc accelerationAngle 175

	velocityToPlayer

	runtime 100
	repeatStart

END

END
